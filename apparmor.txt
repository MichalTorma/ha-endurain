#include <tunables/global>

profile endurian flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other add-on files
  /data/** rw,

  # Start new shells and basic commands
  /bin/bash ix,
  /usr/bin/env ix,
  /bin/echo ix,
  /bin/cat ix,
  /bin/ls ix,
  /usr/bin/find ix,

  # Access to Home Assistant directories
  /config/** rw,
  /share/** rw,
  /ssl/** r,

  # Python and application files
  /usr/bin/python3 ix,
  /opt/venv/bin/python ix,
  /opt/venv/** r,
  /app/** ix,
  /usr/lib/python3*/** r,
  /usr/local/lib/python3*/** r,

  # Network access
  network,

  # System files
  /etc/passwd r,
  /etc/group r,
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/ssl/certs/** r,
  /etc/ca-certificates/** r,
  /usr/share/ca-certificates/** r,

  # Process management
  /proc/*/stat r,
  /proc/*/status r,
  /proc/sys/kernel/pid_max r,
  @{PROC}/sys/net/core/rmem_default r,
  @{PROC}/sys/net/core/rmem_max r,
  @{PROC}/sys/net/core/wmem_default r,
  @{PROC}/sys/net/core/wmem_max r,

  # Device access
  /dev/null rw,
  /dev/zero r,
  /dev/random r,
  /dev/urandom r,

  # Allow application to write logs
  /var/log/endurian/** rw,

  # Application uploads and data
  /share/endurian/** rw,
  /config/endurian/** rw,

  # Deny some dangerous paths
  deny /proc/kcore r,
  deny /proc/sysrq-trigger w,
  deny /sys/kernel/security/** rw,
}
